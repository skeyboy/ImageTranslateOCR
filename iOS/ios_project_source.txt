import Foundation

/// OCR 识别结果
struct RecognizedText: Identifiable, Hashable {
    let id = UUID()
    let text: String
    let confidence: Float
    let boundingBox: CGRect
    
    // 转换到图片坐标的矩形
    var rect: CGRect { boundingBox }
    
    init(text: String, confidence: Float = 1.0, boundingBox: CGRect) {
        self.text = text
        self.confidence = confidence
        self.boundingBox = boundingBox
    }
    
    func hash(into hasher: inout Hasher) {
        hasher.combine(id)
    }
    
    static func == (lhs: RecognizedText, rhs: RecognizedText) -> Bool {
        lhs.id == rhs.id
    }
}

/// 遮罩模式
enum MaskMode: String, CaseIterable, Identifiable {
    case rect
    case precise
    case hybrid
    
    var id: String { rawValue }
    
    var displayName: String {
        switch self {
        case .rect: return "矩形"
        case .precise: return "精确"
        case .hybrid: return "混合"
        }
    }
    
    var description: String {
        switch self {
        case .rect: return "快速，适合纯色背景"
        case .precise: return "精确，适合复杂背景"
        case .hybrid: return "自动选择最佳模式"
        }
    }
}

import Vision
import UIKit

/// OCR 文字识别服务
class OCRService {
    
    /// 使用 Vision 框架进行文字识别
    /// - Parameter image: 输入图片
    /// - Returns: 识别到的文字及位置信息
    func recognize(in image: UIImage) async throws -> [RecognizedText] {
        guard let cgImage = image.cgImage else {
            throw OCRError.invalidImage
        }
        
        let results = try await performRecognition(on: cgImage, imageSize: image.size)
        return results.filter { !$0.text.trimmingCharacters(in: .whitespaces).isEmpty }
    }
    
    /// 执行 Vision 识别请求
    private func performRecognition(
        on cgImage: CGImage,
        imageSize: CGSize
    ) async throws -> [RecognizedText] {
        
        try await withCheckedThrowingContinuation { continuation in
            
            let request = VNRecognizeTextRequest { request, error in
                if let error = error {
                    continuation.resume(throwing: error)
                    return
                }
                
                let results = self.parseResults(
                    request.results as? [VNRecognizedTextObservation] ?? [],
                    imageSize: imageSize
                )
                continuation.resume(returning: results)
            }
            
            // 配置识别参数
            request.recognitionLanguages = ["zh-Hans", "zh-Hant", "en"]
            request.recognitionLevel = .accurate
            request.usesLanguageCorrection = true
            request.minimumTextHeight = 0.01  // 最小文字高度比例
            
            let handler = VNImageRequestHandler(cgImage: cgImage, options: [:])
            
            do {
                try handler.perform([request])
            } catch {
                continuation.resume(throwing: error)
            }
        }
    }
    
    /// 解析识别结果
    private func parseResults(
        _ observations: [VNRecognizedTextObservation],
        imageSize: CGSize
    ) -> [RecognizedText] {
        
        observations.compactMap { observation in
            guard let candidate = observation.topCandidates(1).first else {
                return nil
            }
            
            // Vision 坐标转换：
            // Vision 使用归一化坐标 (0,0)为左下角
            // UIKit 使用像素坐标 (0,0)为左上角
            let boundingBox = observation.boundingBox
            let rect = CGRect(
                x: boundingBox.origin.x * imageSize.width,
                y: (1 - boundingBox.origin.y - boundingBox.height) * imageSize.height,
                width: boundingBox.width * imageSize.width,
                height: boundingBox.height * imageSize.height
            )
            
            return RecognizedText(
                text: candidate.string,
                confidence: candidate.confidence,
                boundingBox: rect
            )
        }
    }
    
    enum OCRError: LocalizedError {
        case invalidImage
        case recognitionFailed(String)
        
        var errorDescription: String? {
            switch self {
            case .invalidImage:
                return "无效的图片"
            case .recognitionFailed(let message):
                return "识别失败：\(message)"
            }
        }
    }
}

@available(iOS 17.4, *)
class TranslateService {
    
    private var translator: Translator?
    
    /// 翻译文字（中文 → 英文）
    func translate(_ text: String) async throws -> String {
        
        let configuration = Translator.Configuration(
            source: .chinese,
            target: .english
        )
        
        let translator = Translator(configuration: configuration)
        
        // 检查语言支持
        let availability = LanguageAvailability()
        let status = try await availability.status(
            from: .chinese,
            to: .english
        )
        
        switch status {
        case .installed:
            break
        case .supported:
            // 需要下载
            try await prepareTranslation()
        default:
            throw TranslateError.languageNotSupported
        }
        
        let response = try await translator.translate(text)
        return response.targetText
    }
    
    private func prepareTranslation() async throws {
        // 下载翻译模型
        let configuration = Translator.Configuration(
            source: .chinese,
            target: .english
        )
        // 等待下载完成
        try await Task.sleep(nanoseconds: 1_000_000_000)
    }
    
    enum TranslateError: LocalizedError {
        case languageNotSupported
        case translationFailed(String)
        
        var errorDescription: String? {
            switch self {
            case .languageNotSupported:
                return "不支持的语言组合"
            case .translationFailed(let message):
                return "翻译失败：\(message)"
            }
        }
    }
}

import UIKit

/// 图片擦除服务
class InpaintService {
    
    /// 擦除图片中的文字区域
    /// - Parameters:
    ///   - image: 原图
    ///   - regions: 文字区域
    ///   - mode: 遮罩模式
    /// - Returns: 擦除后的图片
    func eraseText(
        from image: UIImage,
        regions: [CGRect],
        mode: MaskMode
    ) async -> UIImage {
        
        await withCheckedContinuation { continuation in
            DispatchQueue.global(qos: .userInitiated).async {
                let result: UIImage
                
                switch mode {
                case .rect:
                    result = OpenCVWrapper.rectErase(image, regions: regions)
                    
                case .precise:
                    result = OpenCVWrapper.preciseErase(image, regions: regions)
                    
                case .hybrid:
                    // 混合模式：根据背景复杂度自动选择
                    if self.isComplexBackground(image, regions: regions) {
                        result = OpenCVWrapper.preciseErase(image, regions: regions)
                    } else {
                        result = OpenCVWrapper.rectErase(image, regions: regions)
                    }
                }
                
                continuation.resume(returning: result)
            }
        }
    }
    
    /// 判断背景是否复杂
    private func isComplexBackground(_ image: UIImage, regions: [CGRect]) -> Bool {
        // 简单判断：区域较大或数量较多时使用精确模式
        let totalArea = regions.reduce(0) { $0 + $1.width * $1.height }
        let imageArea = image.size.width * image.size.height
        return totalArea / imageArea > 0.3 || regions.count > 5
    }
}

/// 图片保存服务
class ImageSaveService {
    
    /// 保存图片到相册
    func saveToPhotoLibrary(_ image: UIImage) async throws {
        let status = await PHPhotoLibrary.requestAuthorization(for: .addOnly)
        
        guard status == .authorized || status == .limited else {
            throw SaveError.noPermission
        }
        
        try await withCheckedThrowingContinuation { (continuation: CheckedContinuation<Void, Error>) in
            PHPhotoLibrary.shared().performChanges {
                PHAssetChangeRequest.creationRequestForAsset(from: image)
            } completionHandler: { success, error in
                if success {
                    continuation.resume()
                } else {
                    continuation.resume(
                        throwing: error ?? SaveError.saveFailed
                    )
                }
            }
        }
    }
    
    enum SaveError: LocalizedError {
        case noPermission
        case saveFailed
        
        var errorDescription: String? {
            switch self {
            case .noPermission:
                return "没有相册访问权限"
            case .saveFailed:
                return "保存失败"
            }
        }
    }
}

import SwiftUI
import Combine

@MainActor
class ImageTranslateViewModel: ObservableObject {
    
    // MARK: - Published 属性
    @Published var originalImage: UIImage?
    @Published var resultImage: UIImage?
    @Published var isProcessing = false
    @Published var statusMessage = "就绪，请选择图片"
    @Published var maskMode: MaskMode = .rect
    
    // MARK: - 服务
    private let ocrService = OCRService()
    private let translateService = LegacyTranslateService() // 兼容所有版本
    private let inpaintService = InpaintService()
    private let saveService = ImageSaveService()
    
    // MARK: - 计算属性
    var canTranslate: Bool {
        originalImage != nil && !isProcessing
    }
    
    // MARK: - 翻译流程
    func translateImage() async throws {
        guard let image = originalImage else { return }
        
        isProcessing = true
        defer { isProcessing = false }
        
        // 1. 缩放图片
        statusMessage = "正在预处理..."
        let scaledImage = image.scaledToMaxSize(1024)
        
        // 2. OCR 识别
        statusMessage = "正在识别文字..."
        let recognizedTexts = try await ocrService.recognize(in: scaledImage)
        
        guard !recognizedTexts.isEmpty else {
            statusMessage = "未识别到文字"
            return
        }
        
        // 3. 翻译
        statusMessage = "正在翻译 \(recognizedTexts.count) 段文字..."
        var translations: [String: String] = [:]
        for text in recognizedTexts {
            translations[text.text] = translateService.translate(text.text)
        }
        
        // 4. 擦除文字
        statusMessage = "正在擦除原文字（\(maskMode.displayName)模式）..."
        let regions = recognizedTexts.map { $0.boundingBox }
        let erasedImage = await inpaintService.eraseText(
            from: scaledImage,
            regions: regions,
            mode: maskMode
        )
        
        // 5. 绘制翻译
        statusMessage = "正在写入翻译..."
        let finalImage = drawTranslations(
            translations: translations,
            recognizedTexts: recognizedTexts,
            on: erasedImage
        )
        
        resultImage = finalImage
        statusMessage = "翻译完成！共处理 \(recognizedTexts.count) 段文字"
    }
    
    // MARK: - 绘制翻译文字
    private func drawTranslations(
        translations: [String: String],
        recognizedTexts: [RecognizedText],
        on image: UIImage
    ) -> UIImage {
        
        let renderer = UIGraphicsImageRenderer(size: image.size)
        
        return renderer.image { context in
            // 绘制原图
            image.draw(at: .zero)
            
            // 绘制每段翻译文字
            for recognizedText in recognizedTexts {
                guard let translated = translations[recognizedText.text] else { continue }
                
                let rect = recognizedText.boundingBox
                let fontSize = calculateFitFontSize(
                    text: translated,
                    maxWidth: rect.width
                )
                
                // 文字属性
                let paragraphStyle = NSMutableParagraphStyle()
                paragraphStyle.alignment = .center
                paragraphStyle.lineBreakMode = .byTruncatingTail
                
                let attributes: [NSAttributedString.Key: Any] = [
                    .font: UIFont.boldSystemFont(ofSize: fontSize),
                    .foregroundColor: UIColor.black,
                    .paragraphStyle: paragraphStyle
                ]
                
                // 计算居中位置
                let textSize = translated.size(withAttributes: attributes)
                let x = rect.midX - textSize.width / 2
                let y = rect.midY - textSize.height / 2
                
                // 绘制背景（可选，让文字更清晰）
                let bgRect = CGRect(
                    x: x - 2,
                    y: y - 1,
                    width: textSize.width + 4,
                    height: textSize.height + 2
                )
                context.cgContext.setFillColor(UIColor.white.withAlphaComponent(0.8).cgColor)
                context.cgContext.fill(bgRect)
                
                // 绘制文字
                translated.draw(
                    at: CGPoint(x: x, y: y),
                    withAttributes: attributes
                )
            }
        }
    }
    
    // MARK: - 计算适合的字号
    private func calculateFitFontSize(text: String, maxWidth: CGFloat) -> CGFloat {
        var size: CGFloat = 30
        
        while size > 8 {
            let attributes: [NSAttributedString.Key: Any] = [
                .font: UIFont.boldSystemFont(ofSize: size)
            ]
            let textWidth = text.size(withAttributes: attributes).width
            
            if textWidth <= maxWidth {
                break
            }
            size -= 1
        }
        
        return size
    }
    
    // MARK: - 保存结果
    func saveResult() {
        guard let image = resultImage else { return }
        
        Task {
            do {
                try await saveService.saveToPhotoLibrary(image)
                statusMessage = "已保存到相册"
            } catch {
                statusMessage = "保存失败：\(error.localizedDescription)"
            }
        }
    }
}

import UIKit

extension UIImage {
    
    /// 缩放图片到最大尺寸
    /// - Parameter maxSize: 最大边长（像素）
    /// - Returns: 缩放后的图片
    func scaledToMaxSize(_ maxSize: CGFloat) -> UIImage {
        let scale = min(
            maxSize / size.width,
            maxSize / size.height,
            1.0
        )
        
        guard scale < 1.0 else { return self }
        
        let newSize = CGSize(
            width: size.width * scale,
            height: size.height * scale
        )
        
        let renderer = UIGraphicsImageRenderer(size: newSize)
        return renderer.image { _ in
            self.draw(in: CGRect(origin: .zero, size: newSize))
        }
    }
    
    /// 裁剪图片
    func cropped(to rect: CGRect) -> UIImage? {
        guard let cgImage = cgImage?.cropping(to: rect) else {
            return nil
        }
        return UIImage(cgImage: cgImage)
    }
}

#import "OpenCVWrapper.h"

// 条件编译 OpenCV
#ifdef HAS_OPENCV
#import <opencv2/opencv.hpp>
#import <opencv2/imgcodecs/ios.h>
#import <opencv2/photo.hpp>
#endif

@implementation OpenCVWrapper

+ (BOOL)isOpenCVLoaded {
#ifdef HAS_OPENCV
    return YES;
#else
    return NO;
#endif
}

+ (UIImage *)rectErase:(UIImage *)image regions:(NSArray<NSValue *> *)regions {
#ifdef HAS_OPENCV
    cv::Mat src;
    UIImageToMat(image, src);
    
    // 创建遮罩
    cv::Mat mask = cv::Mat::zeros(src.size(), CV_8UC1);
    
    for (NSValue *rectValue in regions) {
        CGRect rect = [rectValue CGRectValue];
        
        // 扩展区域
        int padding = 4;
        cv::Rect cvRect(
            MAX(0, (int)rect.origin.x - padding),
            MAX(0, (int)rect.origin.y - padding),
            MIN((int)rect.size.width + 2 * padding, src.cols - (int)rect.origin.x),
            MIN((int)rect.size.height + 2 * padding, src.rows - (int)rect.origin.y)
        );
        
        if (cvRect.width > 0 && cvRect.height > 0) {
            cv::rectangle(mask, cvRect, cv::Scalar(255), -1);
        }
    }
    
    // Inpaint 修复
    cv::Mat result;
    cv::inpaint(src, mask, result, 5.0, cv::INPAINT_TELEA);
    
    return MatToUIImage(result);
#else
    return image;
#endif
}

+ (UIImage *)preciseErase:(UIImage *)image regions:(NSArray<NSValue *> *)regions {
#ifdef HAS_OPENCV
    cv::Mat src;
    UIImageToMat(image, src);
    
    // 转灰度图
    cv::Mat gray;
    cv::cvtColor(src, gray, cv::COLOR_BGR2GRAY);
    
    // 创建遮罩
    cv::Mat mask = cv::Mat::zeros(src.size(), CV_8UC1);
    
    for (NSValue *rectValue in regions) {
        CGRect rect = [rectValue CGRectValue];
        
        cv::Rect cvRect(
            MAX(0, (int)rect.origin.x),
            MAX(0, (int)rect.origin.y),
            MIN(src.cols - (int)rect.origin.x, (int)rect.size.width),
            MIN(src.rows - (int)rect.origin.y, (int)rect.size.height)
        );
        
        if (cvRect.width <= 0 || cvRect.height <= 0) continue;
        
        // 提取 ROI
        cv::Mat roiGray = gray(cvRect);
        cv::Mat roiMask = mask(cvRect);
        cv::Mat binary;
        
        // 自适应阈值提取文字
        cv::adaptiveThreshold(
            roiGray, binary, 255,
            cv::ADAPTIVE_THRESH_GAUSSIAN_C,
            cv::THRESH_BINARY_INV,
            15, 4.0
        );
        
        binary.copyTo(roiMask);
    }
    
    // 膨胀连接笔画
    cv::Mat kernel = cv::getStructuringElement(cv::MORPH_ELLIPSE, cv::Size(5, 5));
    cv::dilate(mask, mask, kernel);
    
    // Inpaint 修复
    cv::Mat result;
    cv::inpaint(src, mask, result, 3.0, cv::INPAINT_TELEA);
    
    return MatToUIImage(result);
#else
    return image;
#endif
}

+ (NSArray<UIImage *> *)batchErase:(NSArray<UIImage *> *)images
                      regionsArray:(NSArray<NSArray<NSValue *> *> *)regionsArray {
    NSMutableArray *results = [NSMutableArray array];
    
    for (NSInteger i = 0; i < images.count; i++) {
        UIImage *image = images[i];
        NSArray *regions = i < regionsArray.count ? regionsArray[i] : @[];
        
        UIImage *result = [self rectErase:image regions:regions];
        [results addObject:result];
    }
    
    return results;
}

@end

# ImageTranslateOCR - iOS

iOS 图片 OCR 文字识别 + 翻译 + 原图覆盖工具

## 功能

- 📷 选择相册图片
- 🔍 Vision 框架 OCR 中文识别
- 🌐 Translation API 翻译（中→英）
- 🎨 OpenCV Inpaint 擦除原文字
- ✍️ UIGraphicsImageRenderer 绘制翻译
- 💾 保存到相册

## 技术栈

| 模块 | 方案 |
|------|------|
| OCR | Vision (VNRecognizeTextRequest) |
| 翻译 | Translation API (iOS 15+) |
| 擦除 | OpenCV Inpaint (Telea) |
| UI | SwiftUI |
| 异步 | async/await |

## 系统要求

- iOS 16.0+
- Xcode 15.0+
- Swift 5.9+
- OpenCV 4.8+（可选，用于擦除功能）

## 安装

1. 安装 OpenCV（二选一）

pod 'OpenCV', '~> 4.8.0'
